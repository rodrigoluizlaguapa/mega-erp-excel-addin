require('dotenv').config(); const express = require('express'); const axios = require('axios'); const NodeCache = require('node-cache'); const cors = require('cors'); const app = express(); const tokenCache = new NodeCache({ stdTTL: 3600 }); const BASE_URL = process.env.MEGA_URL; const USER = process.env.MEGA_USER; const PASS = process.env.MEGA_PASS; const TENANT_ID = process.env.MEGA_TENANT; const VERSAO = "v8.0 - Node.js"; app.use(cors()); app.use(express.json()); // ───────────────────────────────────────────── // AUTH COM CACHE // ───────────────────────────────────────────── async function getMegaToken() { const cached = tokenCache.get('mega_token'); if (cached) return cached; console.log('🔄 Autenticando no Mega ERP...'); const response = await axios.post( `${BASE_URL}/api/auth/signin`, { userName: USER, password: PASS, tenant: TENANT_ID }, { headers: { 'TenantId': TENANT_ID }, timeout: 10000 } ); const token = response.data.accessToken; tokenCache.set('mega_token', token); console.log('✅ Token obtido e cacheado.'); return token; } async function chamarMega(url, token) { return axios.get(url, { headers: { 'Authorization': `Bearer ${token}`, 'TenantId': TENANT_ID, 'Accept': 'application/json' }, timeout: 20000 }).then(res => res.data); } // ───────────────────────────────────────────── // ROTA PRINCIPAL — despacha por action // ───────────────────────────────────────────── app.get('/', async (req, res) => { const action = req.query.action; console.log(`\n📥 action=${action} | ${JSON.stringify(req.query)}`); try { const token = await getMegaToken(); // ── LIST: página de cabeçalhos ────────────────────────── if (action === 'list') { const { inicio, fim, page = 1, limit = 100 } = req.query; if (!inicio || !fim) { return res.status(400).json({ success: false, error: "Parâmetros 'inicio' e 'fim' são obrigatórios." }); } const url = `${BASE_URL}/api/contabilidadelancamentos/lancamento` + `?DataInicial=${inicio}&DataFinal=${fim}` + `&Page=${page}&Limit=${limit}`; console.log(`🔍 LIST página ${page}: ${url}`); const data = await chamarMega(url, token); const lancamentos = Array.isArray(data) ? data : data.items || data.Items || data.data || []; console.log(`✅ ${lancamentos.length} lançamentos na página ${page}.`); return res.json({ success: true, versao: VERSAO, data: lancamentos }); } // ── DETALHE: itens de um lançamento específico ────────── if (action === 'detalhe') { const { id } = req.query; if (!id) { return res.status(400).json({ success: false, error: "Parâmetro 'id' é obrigatório." }); } const expandParams = [ "Itens", "Itens.Documento", "Itens.Historico", "Itens.Agente", "RateiosCentroCusto", "RateiosCentroCusto.CentroCusto", "RateiosProjeto", "RateiosProjeto.Projeto" ].join(","); const url = `${BASE_URL}/api/contabilidadelancamentos/lancamento/${id}?Expand=${expandParams}`; console.log(`🔍 DETALHE id=${id}`); const data = await chamarMega(url, token); console.log(`✅ Detalhe obtido para id=${id}.`); return res.json({ success: true, versao: VERSAO, data }); } // ── AGENTES: lista de agentes ─────────────────────────── if (action === 'agentes') { const url = `${BASE_URL}/api/agentes/agente`; console.log(`🔍 AGENTES: ${url}`); const data = await chamarMega(url, token); const agentes = Array.isArray(data) ? data : data.items || data.Items || data.data || []; console.log(`✅ ${agentes.length} agentes obtidos.`); return res.json({ success: true, versao: VERSAO, data: agentes }); } // ── ACTION DESCONHECIDA ───────────────────────────────── return res.status(400).json({ success: false, error: `Action desconhecida: '${action}'` }); } catch (error) { console.error(`❌ Erro [action=${action}]:`, error.message); return res.status(500).json({ success: false, versao: VERSAO, error: error.message }); } }); // ───────────────────────────────────────────── // POST — inclusão de agentes // ───────────────────────────────────────────── app.post('/', async (req, res) => { const { action, agentes } = req.body; console.log(`\n📥 POST action=${action} | ${agentes?.length} registros`); if (action !== 'incluirAgentes') { return res.status(400).json({ success: false, error: `Action POST desconhecida: '${action}'` }); } try { const token = await getMegaToken(); let incluidos = 0; let erros = 0; const resultados = []; for (const ag of agentes) { try { await axios.post( `${BASE_URL}/api/agentes/agente`, ag, { headers: { 'Authorization': `Bearer ${token}`, 'TenantId': TENANT_ID, 'Content-Type': 'application/json' }, timeout: 10000 } ); incluidos++; resultados.push({ status: "OK" }); console.log(`✅ Agente incluído: ${ag.NOME_RAZAO}`); } catch (err) { erros++; const msg = err.response?.data?.message || err.message; resultados.push({ status: `ERRO: ${msg}` }); console.error(`❌ Erro ao incluir ${ag.NOME_RAZAO}: ${msg}`); } } return res.json({ success: true, versao: VERSAO, incluidos, erros, resultados }); } catch (error) { console.error('❌ Erro geral no POST:', error.message); return res.status(500).json({ success: false, versao: VERSAO, error: error.message }); } }); // ───────────────────────────────────────────── // HEALTH CHECK // ───────────────────────────────────────────── app.get('/health', (req, res) => { res.json({ status: 'UP', versao: VERSAO, time: new Date().toISOString() }); }); // ───────────────────────────────────────────── // START // ───────────────────────────────────────────── const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`\n=========================================`); console.log(`🚀 LG CFO Server: http://localhost:${PORT}`); console.log(`📡 Versão: ${VERSAO}`); console.log(`=========================================\n`); });
